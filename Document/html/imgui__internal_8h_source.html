218 // Helper: ImPool<>. Basic keyed storage for contiguous instances, slow/amortized insertion, O(1) indexable, O(Log N) queries by ID over a dense/hot buffer,
219 // Honor constructor/destructor. Add/remove invalidate all pointers. Indexes have the same lifetime as the associated object.
220 typedef int ImPoolIdx;
221 template<typename T>
222 struct IMGUI_API ImPool
223 {
224  ImVector<T> Data; // Contiguous data
225  ImGuiStorage Map; // ID->Index
226  ImPoolIdx FreeIdx; // Next free idx to use
227 
228  ImPool() { FreeIdx = 0; }
229  ~ImPool() { Clear(); }
230  T* GetByKey(ImGuiID key) { int idx = Map.GetInt(key, -1); return (idx != -1) ? &Data[idx] : NULL; }
231  T* GetByIndex(ImPoolIdx n) { return &Data[n]; }
232  ImPoolIdx GetIndex(const T* p) const { IM_ASSERT(p >= Data.Data && p < Data.Data + Data.Size); return (ImPoolIdx)(p - Data.Data); }
233  T* GetOrAddByKey(ImGuiID key) { int* p_idx = Map.GetIntRef(key, -1); if (*p_idx != -1) return &Data[*p_idx]; *p_idx = FreeIdx; return Add(); }
234  void Clear() { for (int n = 0; n < Map.Data.Size; n++) { int idx = Map.Data[n].val_i; if (idx != -1) Data[idx].~T(); } Map.Clear(); Data.clear(); FreeIdx = 0; }
235  T* Add() { int idx = FreeIdx; if (idx == Data.Size) { Data.resize(Data.Size + 1); FreeIdx++; } else { FreeIdx = *(int*)&Data[idx]; } IM_PLACEMENT_NEW(&Data[idx]) T(); return &Data[idx]; }
236  void Remove(ImGuiID key, const T* p) { Remove(key, GetIndex(p)); }
237  void Remove(ImGuiID key, ImPoolIdx idx) { Data[idx].~T(); *(int*)&Data[idx] = FreeIdx; FreeIdx = idx; Map.SetInt(key, -1); }
238  void Reserve(int capacity) { Data.reserve(capacity); Map.Data.reserve(capacity); }
239  int GetSize() const { return Data.Size; }
240 };

420 struct IMGUI_API ImRect
421 {
422  ImVec2 Min; // Upper-left
423  ImVec2 Max; // Lower-right
424 
425  ImRect() : Min(FLT_MAX,FLT_MAX), Max(-FLT_MAX,-FLT_MAX) {}
426  ImRect(const ImVec2& min, const ImVec2& max) : Min(min), Max(max) {}
427  ImRect(const ImVec4& v) : Min(v.x, v.y), Max(v.z, v.w) {}
428  ImRect(float x1, float y1, float x2, float y2) : Min(x1, y1), Max(x2, y2) {}
429 
430  ImVec2 GetCenter() const { return ImVec2((Min.x + Max.x) * 0.5f, (Min.y + Max.y) * 0.5f); }
431  ImVec2 GetSize() const { return ImVec2(Max.x - Min.x, Max.y - Min.y); }
432  float GetWidth() const { return Max.x - Min.x; }
433  float GetHeight() const { return Max.y - Min.y; }
434  ImVec2 GetTL() const { return Min; } // Top-left
435  ImVec2 GetTR() const { return ImVec2(Max.x, Min.y); } // Top-right
436  ImVec2 GetBL() const { return ImVec2(Min.x, Max.y); } // Bottom-left
437  ImVec2 GetBR() const { return Max; } // Bottom-right
438  bool Contains(const ImVec2& p) const { return p.x >= Min.x && p.y >= Min.y && p.x < Max.x && p.y < Max.y; }
439  bool Contains(const ImRect& r) const { return r.Min.x >= Min.x && r.Min.y >= Min.y && r.Max.x <= Max.x && r.Max.y <= Max.y; }
440  bool Overlaps(const ImRect& r) const { return r.Min.y < Max.y && r.Max.y > Min.y && r.Min.x < Max.x && r.Max.x > Min.x; }
441  void Add(const ImVec2& p) { if (Min.x > p.x) Min.x = p.x; if (Min.y > p.y) Min.y = p.y; if (Max.x < p.x) Max.x = p.x; if (Max.y < p.y) Max.y = p.y; }
442  void Add(const ImRect& r) { if (Min.x > r.Min.x) Min.x = r.Min.x; if (Min.y > r.Min.y) Min.y = r.Min.y; if (Max.x < r.Max.x) Max.x = r.Max.x; if (Max.y < r.Max.y) Max.y = r.Max.y; }
443  void Expand(const float amount) { Min.x -= amount; Min.y -= amount; Max.x += amount; Max.y += amount; }
444  void Expand(const ImVec2& amount) { Min.x -= amount.x; Min.y -= amount.y; Max.x += amount.x; Max.y += amount.y; }
445  void Translate(const ImVec2& d) { Min.x += d.x; Min.y += d.y; Max.x += d.x; Max.y += d.y; }
446  void TranslateX(float dx) { Min.x += dx; Max.x += dx; }
447  void TranslateY(float dy) { Min.y += dy; Max.y += dy; }
448  void ClipWith(const ImRect& r) { Min = ImMax(Min, r.Min); Max = ImMin(Max, r.Max); } // Simple version, may lead to an inverted rectangle, which is fine for Contains/Overlaps test but not for display.
449  void ClipWithFull(const ImRect& r) { Min = ImClamp(Min, r.Min, r.Max); Max = ImClamp(Max, r.Min, r.Max); } // Full version, ensure both points are fully clipped.
450  void Floor() { Min.x = (float)(int)Min.x; Min.y = (float)(int)Min.y; Max.x = (float)(int)Max.x; Max.y = (float)(int)Max.y; }
451  bool IsInverted() const { return Min.x > Max.x || Min.y > Max.y; }

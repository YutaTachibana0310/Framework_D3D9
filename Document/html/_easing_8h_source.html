51  template <class T>
52  static T EaseValue(float time, const T& start, const T& end, EaseType type)
53  {
54  if (time >= 1.0f)
55  return end;
56 
57  if (time <= 0.0f)
58  return start;
59 
60  T result;
61 
62  if (type == EaseType::Linear)
63  {
64  result = (end - start) * time + start;
65  }
66  else
67  {
68  float c[4];
69  GetController(c, type);
70 
71  float x = CubicBezier(c[0], c[1], c[2], c[3], time);
72  result = (end - start) * CubicBezier(c[0], c[1], c[2], c[3], x) + start;
73  }
74 
75  return result;
76  }
